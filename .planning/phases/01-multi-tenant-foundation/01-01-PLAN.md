```prisma
model free_zones {
  id             String   @id @default(uuid())
  code           String   @unique  // e.g., "JAFZA", "DAFZA", "DIFC"
  name           String   // "Jebel Ali Free Zone"
  nameArabic     String?  // Arabic name for bilingual support
  emirate        String   // "Dubai", "Abu Dhabi", etc.
  isDesignated   Boolean  @default(false) // VAT Designated Zone status (Cabinet Decision No. 59)
  vatTreatment   String   @default("STANDARD_RATE") // "OUTSIDE_SCOPE", "STANDARD_RATE"
  isCtQualifying Boolean  @default(false) // Corporate Tax Qualifying Free Zone
  isActive       Boolean  @default(true)
  createdAt      DateTime @default(now())
  updatedAt      DateTime @updatedAt

  @@index([isDesignated])
  @@index([emirate])
  @@index([isActive])
  @@map("free_zones")
}
```

2. **industry_codes model** - Industry classification reference:
```prisma
model industry_codes {
  id             String   @id @default(uuid())
  dedCode        String?  @unique // Dubai DED activity code
  isicCode       String   // ISIC Rev 4 code for international reporting
  name           String
  nameArabic     String?
  category       String   // "Trading", "Services", "Manufacturing", "Construction", etc.
  vatDefaultRate String   @default("STANDARD") // Default VAT treatment: "STANDARD", "ZERO", "EXEMPT"
  hasSpecialRules Boolean @default(false) // Indicates industry-specific compliance rules
  isActive       Boolean  @default(true)
  createdAt      DateTime @default(now())
  updatedAt      DateTime @updatedAt

  @@index([isicCode])
  @@index([category])
  @@index([isActive])
  @@map("industry_codes")
}
```

```prisma
model tenant_compliance_config {

  industryName      String?  // Denormalized for display

  vatGroupMember    Boolean  @default(false)
```

```typescript
  console.log('Seeding UAE free zones...');

  for (const zone of freeZones) {
    await prisma.free_zones.upsert({
      where: { code: zone.code },
      update: {
        name: zone.name,
        emirate: zone.emirate,
        isDesignated: zone.isDesignated,
        vatTreatment: zone.isDesignated ? 'OUTSIDE_SCOPE' : 'STANDARD_RATE',
        isCtQualifying: zone.isCtQualifying,
```
